package main

import (
	"./bio"
	"fmt"
	"os"
)

func main() {
	args := os.Args

	sequence := bio.SequenceFromRosalindFile(args[1])

	as := bio.CountBaseOccurence(sequence, "A")
	cs := bio.CountBaseOccurence(sequence, "C")
	gs := bio.CountBaseOccurence(sequence, "G")
	ts := bio.CountBaseOccurence(sequence, "T")

	fmt.Printf("%d %d %d %d\n", as, cs, gs, ts)
}
